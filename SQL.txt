 برای نشان دادن کارمندی که فامیلش احمدی است 

SELECT ln FROM bakhsh as b INNER JOIN karmandan as k ON b.dcode=k.dcode WHERE k.ln="ahmadi";

برای نام و نام خانوادگی کارمند های که برای بیشتر از 2 پروژه کار میکنند 

SELECT k.ln,k.fn,COUNT(w.pcode)as tadad FROM work as w INNER JOIN karmandan as k ON w.ecode=k.code GROUP BY k.ln,k.fn,w.pcode HAVING tadad >=2;

_____________________________________________________________________________________________________________________

نام و  نام خانوادگی کارمندانی که رو پروژه ای که تو نیشابور هست کار میکنند 

SELECT ln,fn FROM work as w INNER JOIN karmandan as k ON w.ecode=k.code INNER JOIN proje as p ON w.pcode = p.code WHERE p.plocation="Neyshabur";

_____________________________________________________________________________________________________________________

نام و نام خانوادگی کارمندانی که رو یک پروژه کار میکنند 

SELECT k.ln,k.fn,COUNT(p.code) FROM work as w INNER JOIN karmandan as k ON w.ecode=k.code INNER JOIN proje as p ON w.pcode=p.code GROUP BY k.ln,k.fn,p.code HAVING p.code >1;

نمایش تمام کارکنان با نام بخش آن‌ها (حتی اگر در بخشی کار نمی‌کنند)

SELECT k.ln,k.fn,b.name FROM karmandan as k LEFT JOIN bakhsh as b ON k.dcode=b.dcode;		

_____________________________________________________________________________________________________________________

نمایش تمام پروژه‌هایی که کارکنانی روی آن کار کرده‌اند، حتی اگر برخی پروژه‌ها هنوز کارمندی ندارند

SELECT k.ln,k.fn,p.name FROM proje as p LEFT JOIN work as w ON p.code=w.pcode LEFT JOIN karmandan AS k ON w.ecode=k.code;

_____________________________________________________________________________________________________________________

نمایش تمام کارمندانی که روی پروژه ای کار میکنند حتی کارمندانی که روی پروژه ای کار نمیکنند را نمایش بده 

SELECT k.ln,k.fn,p.name FROM proje as p RIGHT JOIN work as w ON p.code=w.pcode RIGHT JOIN karmandan AS k ON w.ecode=k.code;

_____________________________________________________________________________________________________________________

 لیست تمام بخش‌ها و تعداد کارکنان هر بخش (حتی بخش‌های بدون کارمند)

SELECT b.name,COUNT(k.code) as tadad FROM bakhsh as b LEFT JOIN karmandan as k ON b.dcode=k.code GROUP BY b.name,k.code;
